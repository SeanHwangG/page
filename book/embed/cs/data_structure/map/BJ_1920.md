```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
set<int> se;

int main(){
  scanf("%d", &n);
  for(int x; n--; ){
    scanf("%d", &x);
    se.insert(x);
  }
  scanf("%d", &n);
  for(int x; n--; ){
    scanf("%d", &x);
    printf("%d\n", se.find(x) != se.end());
  }
}
```

```py
input()
li = list(sorted(map(int, input().split())))
input()
def binary_search(li, x):
  lo, hi = 0, len(li) - 1
  while lo < hi:
    mi = (hi + lo) // 2
    if li[mi] < x:
      lo = mi + 1
    else:
      hi = mi
  return li[lo] == x
for n in map(int, input().split()):
  print(1 if binary_search(li, n) else 0)
```
